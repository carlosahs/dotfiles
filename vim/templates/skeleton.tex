% jub0bs code for highlighting numbers in code block
% extracted from: https://tex.stackexchange.com/questions/34896/coloring-digits-with-the-listings-package
\newcommand\digitstyle{\color{numbers}}
\makeatletter
\newcommand{\ProcessDigit}[1] {
  \ifnum\lst@mode=\lst@Pmode\relax%
    {\digitstyle #1}%
  \else
    #1%
  \fi
}
\makeatother

\lstdefinestyle{mystyle}{

  commentstyle = \color{comments},

  numbersep = 5pt,
  showspaces = false,

  literate = {-}{{{\color{operators}-}}}1

             {\%}{{{\color{operators}\%}}}1

             {0}{{{\ProcessDigit{0}}}}1

             {2}{{{\ProcessDigit{2}}}}1

             {5}{{{\ProcessDigit{5}}}}1
             {6}{{{\ProcessDigit{6}}}}1
             {7}{{{\ProcessDigit{7}}}}1
             {8}{{{\ProcessDigit{8}}}}1

}
